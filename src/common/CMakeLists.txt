set(COMMON_FILES_BASE
    AppConstants.h
)

# 创建带前缀的文件列表Relative
set(RELATIVE_COMMON_SOURCES "")
set(COMMON_SOURCES "")
foreach(SOURCES_FILE ${COMMON_FILES_BASE})
    list(APPEND RELATIVE_COMMON_SOURCES "src/common/${SOURCES_FILE}")
    list(APPEND COMMON_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}/${SOURCES_FILE}")
endforeach()

# 导出变量到父作用域
set(RELATIVE_COMMON_SOURCES ${RELATIVE_COMMON_SOURCES} PARENT_SCOPE)
set(COMMON_SOURCES ${COMMON_SOURCES} PARENT_SCOPE)
