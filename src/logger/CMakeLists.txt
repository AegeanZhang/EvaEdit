set(LOGGER_FILES_BASE
    Logger.h
    Logger.cpp
    QmlLoggerWrapper.h
)

# 创建带前缀的文件列表Relative
set(RELATIVE_LOGGER_SOURCES "")
set(LOGGER_SOURCES "")
foreach(SOURCES_FILE ${LOGGER_FILES_BASE})
    list(APPEND RELATIVE_LOGGER_SOURCES "src/logger/${SOURCES_FILE}")
    list(APPEND LOGGER_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}/${SOURCES_FILE}")
endforeach()

# 导出变量到父作用域
set(RELATIVE_LOGGER_SOURCES ${RELATIVE_LOGGER_SOURCES} PARENT_SCOPE)
set(LOGGER_SOURCES ${LOGGER_SOURCES} PARENT_SCOPE)
