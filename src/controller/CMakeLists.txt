set(CONTROLLER_FILES_BASE
    TabController.h
    TabController.cpp
)

# 创建带前缀的文件列表
set(RELATIVE_CONTROLLER_SOURCES "")
set(CONTROLLER_SOURCES "")
foreach(SOURCES_FILE ${CONTROLLER_FILES_BASE})
    list(APPEND RELATIVE_CONTROLLER_SOURCES "src/controller/${SOURCES_FILE}")
    list(APPEND CONTROLLER_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}/${SOURCES_FILE}")
endforeach()

# 导出变量到父作用域
set(RELATIVE_CONTROLLER_SOURCES ${RELATIVE_CONTROLLER_SOURCES} PARENT_SCOPE)
set(CONTROLLER_SOURCES ${CONTROLLER_SOURCES} PARENT_SCOPE)
